00080     int64 best_est = kint64max;
00081     int64 best_lct = kint64max;
00082     int support = -1;
00083     int refuted = 0;
00084     for (int i = 0; i < size_; ++i) {
00085       IntervalVar* const v = vars_[i];
00086       if (v->MayBePerformed() && v->StartMax() > v->StartMin()) {
00087         if (v->StartMin() >= markers_[i] &&
00088                   (v->StartMin() < best_est ||
00089                    (v->StartMin() == best_est && v->EndMax() < best_lct))) {
00090           best_est = v->StartMin();
00091           best_lct = v->EndMax();
00092           support = i;
00093         } else {
00094           refuted++;
00095         }
00096       }
00097     }
00098     // TODO(user) : remove this crude quadratic loop with
00099     // reversibles range reduction.
00100     if (support == -1) {
00101       if (refuted == 0) {
00102         return NULL;
00103       } else {
00104         s->Fail();
00105       }
00106     }
00107     return s->RevAlloc(new ScheduleOrPostpone(vars_[support],
00108                                               vars_[support]->StartMin(),
00109                                               &markers_[support]));
00110   }

00161 class RankFirstSequences : public DecisionBuilder {
00162  public:
00163   RankFirstSequences(const Sequence* const * sequences, int size)
00164       : sequences_(new Sequence*[size]), size_(size) {
00165   memcpy(sequences_.get(), sequences, size_ * sizeof(*sequences));
00166       }
00167   virtual ~RankFirstSequences() {}

00169   virtual Decision* Next(Solver* const s) {
00170     Sequence* seq = NULL;
00171     int64 slack = kint64max;
00172     int64 best_hmin = kint64max;
00173     for (int i = 0; i < size_; ++i) {
00174       Sequence* se = sequences_[i];
00175       if (se->NotRanked() > 0) {
00176         int64 hmin, hmax, dmin, dmax;
00177         se->HorizonRange(&hmin, &hmax);
00178         se->DurationRange(&dmin, &dmax);
00179         const int64 current_slack = (hmax - hmin - dmax);
00180         if (current_slack < slack) {
00181           slack = current_slack;
00182           seq = se;
00183           int64 hmin, hmax;
00184           se->ActiveHorizonRange(&hmin, &hmax);
00185           best_hmin = hmin;
00186         } else if (current_slack == slack) {
00187           int64 hmin, hmax;
00188           se->ActiveHorizonRange(&hmin, &hmax);
00189           if (hmin < best_hmin) {
00190             seq = se;
00191             best_hmin = hmin;
00192           }
00193         }
00194       }
00195     }
00196     if (seq != NULL) {
00197       seq->ComputePossibleRanks();
00198       int index = -1;
00199       int64 smin = kint64max;
00200       for (int i = 0; i < seq->size(); ++i) {
00201         if (seq->PossibleFirst(i)) {
00202           IntervalVar* t = seq->Interval(i);
00203           if (t->MayBePerformed() && t->StartMin() < smin) {
00204             index = i;
00205             smin = t->StartMin();
00206           }
00207         }
00208       }
00209       if (index == -1) {
00210         s->Fail();
00211       }
00212       CHECK_NE(-1, index);
00213       return s->RevAlloc(new TryRankFirst(seq, index));
00214     }
00215     return NULL;
00216   }
